1 function AudioManager(_game) {
  2 
  3 	this.audioTracks = new Array();
  4 	this.game = _game;
  5 	this.muted = false;
  6 
  7 	var myAudio = document.createElement('audio');
  8 	this.canPlayMp3 = "" != myAudio.canPlayType('audio/mpeg');
  9 	this.canPlayOgg = "" != myAudio.canPlayType('audio/ogg; codecs="vorbis"');
 10 
 11 	this.loadingStatus = function() {
 12 
 13 		var total = 0;
 14 		var countReady = 0;
 15 
 16 		for (key in this.audioTracks) {
 17 			total++;
 18 			var audio = this.audioTracks[key];
 19 
 20 			this.game.debug.warn(key + ' audio state : ' + audio.readyState);
 21 			if (audio.readyState >= audio.HAVE_CURRENT_DATA) {
 22 				countReady++;
 23 			}
 24 		}
 25 
 26 		if (countReady == 0) {
 27 			return 0;
 28 		}
 29 
 30 		return countReady / total;
 31 	};
 32 
 33 	/*
 34 	 * register audio sources
 35 	 */
 36 	this.register = function(_name, _loop) {
 37 
 38 		if (!this.audioTracks[_name]) {
 39 
 40 			var myLocalAudio = new Audio('');
 41 
 42 			if (_loop) {
 43 				myLocalAudio.loop = _loop;
 44 				myLocalAudio.addEventListener('ended', function() {
 45 					this.currentTime = 0;
 46 				}, false);
 47 			}
 48 
 49 			myLocalAudio.autoplay = false;
 50 
 51 			var source = false;
 52 			if (this.canPlayMp3) {
 53 				source = _name + '.mp3';
 54 			} else {
 55 				if (this.canPlayOgg) {
 56 					source = _name + '.ogg';
 57 				}
 58 			}
 59 
 60 			if (source) {
 61 				myLocalAudio.src = source;
 62 				myLocalAudio.load();
 63 				this.audioTracks[_name] = myLocalAudio;
 64 			} else {
 65 				this.game.debug
 66 						.warn('browser not able to play MP3 or OGG');
 67 			}
 68 
 69 		}
 70 	};
 71 
 72 	this.play = function(_name) {
 73 		if (this.audioTracks[_name]) {
 74 			var audio = this.audioTracks[_name];
 75 			if (audio.isFlash) {
 76 				audio.playFlash(audio.src);
 77 			} else {
 78 				audio.play();
 79 			}
 80 
 81 		} else {
 82 			this.game.debug.warn('audio :' + _name + ' not registered');
 83 		}
 84 	};
 85 
 86 	this.mute = function() {
 87 		this.muted = true;
 88 		for (key in this.audioTracks) {
 89 			var audio = this.audioTracks[key];
 90 			audio.muted = true;
 91 		}
 92 	};
 93 
 94 	this.unmute = function() {
 95 		this.muted = false;
 96 		for (key in this.audioTracks) {
 97 			var audio = this.audioTracks[key];
 98 			audio.muted = false;
 99 		}
100 	};
101 
102 	this.setVolume = function(_volume) {
103 		for (key in this.audioTracks) {
104 			var audio = this.audioTracks[key];
105 			audio.volume = _volume;
106 		}
107 	};
108 
109 }
