1 /**
  2  * @class ProgressBar
  3  * @param _id
  4  * @param _x
  5  * @param _y
  6  * @param _width
  7  * @param _height
  8  * @param _position
  9  * @param _max
 10  *
 11  * change position to move the progress bar
 12  */
 13 function ProgressBar(_id, _x, _y, _width, _height, _position, _max ) {
 14 
 15 	this.from = Rectangle;
 16 	this.from();
 17 	this.superUpdate = this.update;
 18 	
 19 	this.id = _id;
 20 	this.x = _x;
 21 	this.y = _y;
 22 	this.width = _width;
 23 	this.height = _height;
 24 	this.position = _position;
 25 	this.max = _max;
 26 	this.defaultWidth = _width - ( 2 * this.border );
 27 
 28 	this.frontRect = new Rectangle(_id + '_front_loading', _x + this.border, _y + this.border,
 29 		this.defaultWidth * (_position/_max), _height - (2* this.border) );
 30 		
 31 	this.defaultWidth = _width - this.border;
 32 	
 33 	this.fillColor = '#555555';
 34 	this.borderColor = '#555555';
 35 	this.frontRect.fillColor = NMSColor.PAPAYA;
 36 	this.frontRect.borderColor = NMSColor.PAPAYA;
 37 	this.frontRect.border = 0;
 38 	
 39 	this.setBarColor = function(_color){
 40 		this.frontRect.fillColor = _color;
 41 		this.frontRect.borderColor = _color;
 42 	}
 43 
 44 	this.update = function(_game, _scene, current) {
 45 		this.superUpdate(_game, _scene, current);
 46 		
 47 		if( this.frontRect.width <= 0 || this.frontRect.height <= 0 ){
 48 			_game.debug.warn('update of not loaded : ' + this.id );
 49 			return;
 50 		}
 51 		
 52 		this.frontRect.fixed = this.fixed;
 53 
 54 		this.frontRect.width = this.defaultWidth * (this.position / this.max);
 55 		
 56 		if( this.frontRect.width < 1 ){
 57 			this.frontRect.width = 1;
 58 		}
 59 		this.frontRect.update(_game, _scene, current);
 60 	};
 61 
 62 }
