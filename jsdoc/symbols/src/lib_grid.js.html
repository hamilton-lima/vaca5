1 // create grid cell that mix rectangle and label
  2 
  3 /**
  4  * @class Grid
  5  * @param _x
  6  * @param _y
  7  * @param _id
  8  * @param _text
  9  * @return
 10  */
 11 function Grid(_id, _x, _y, _width, _height, _lines, _columns, _border) {
 12 
 13 	this.from = GridLayout;
 14 	this.from();
 15 
 16 	this.x = _x;
 17 	this.y = _y;
 18 	this.width = _width;
 19 	this.height = _height;
 20 	this.lines = _lines;
 21 	this.columns = _columns;
 22 	this.border = _border;
 23 	this.fontSize = 12;
 24 	
 25 	this.setHeaderColor = NMSColor.FENNEL;
 26 	this.setDataColor = NMSColor.ENDIVE;
 27 	
 28 	this.id = _id;
 29 	
 30 	/**
 31 	 * _data must have nodes header/data
 32 	 * header must be an array of strings
 33 	 * and data an array of array of strings
 34 	 * 
 35 	 * @param _data
 36 	 * @return
 37 	 */
 38 	this.setData = function(_game, current, _data){
 39 		var headerArray = _data['header'];
 40 		var dataArray = _data['data'];
 41 		
 42 		if( !headerArray || !dataArray ){
 43 			_game.debug.error('no data or header was set');
 44 		} else {
 45 			
 46 			for(key in headerArray){
 47 				var foo = new Label(0,0, 'header_' + key, headerArray[key]);
 48 				foo.size = this.fontSize;
 49 				foo.backgroundColor = current.setHeaderColor;
 50 				current.add( foo );
 51 			}
 52 
 53 			for(key in dataArray){
 54 				for(innerKey in dataArray[key]){
 55 					var label = new Label(0,0, 'data_' + key + '_' + innerKey, dataArray[key][innerKey]);
 56 					label.backgroundColor = current.setDataColor;
 57 					label.size = this.fontSize;
 58 					current.add( label );
 59 				}
 60 			}
 61 		}
 62 		
 63 	};
 64 	
 65 
 66 }
