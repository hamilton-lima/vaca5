1 function ViewportManager(_target,_left,_right,_top,_bottom) {
  2 
  3 	this.followTarget = _target;
  4 	this.leftLimit = _left;
  5 	this.rightLimit = _right;
  6 	this.topLimit = _top;
  7 	this.bottomLimit = _bottom;
  8 
  9 	/**
 10 	* try to set the followtarget in the middle of the screen
 11 	* calculate the necessary shift to move to this position
 12 	* and check if can move in that direction
 13 	*/
 14 	this.update = function(_game) {
 15 	
 16 		if( this.followTarget.width <= 0 || this.followTarget.height <= 0
 17 			|| this.leftLimit.width <= 0 || this.leftLimit.height <= 0 			
 18 			|| this.rightLimit.width <= 0 || this.rightLimit.height <= 0 			
 19 			|| this.topLimit.width <= 0 || this.topLimit.height <= 0 			
 20 			|| this.bottomLimit.width <= 0 || this.bottomLimit.height <= 0 ){
 21 			
 22 			_game.debug.warn('progress bar update with not ready objects.');
 23 			return;
 24 		}
 25 	
 26 		var x = this.followTarget.x - ((_game.viewport.width - this.followTarget.width )/2 );
 27 		var y = this.followTarget.y - ((_game.viewport.height - this.followTarget.height )/2);		
 28 
 29 		_game.viewport.x = x;
 30 		_game.viewport.y = y;
 31 		
 32 		// check the colision with the limits to restrict the viewport movement
 33 		
 34 		// check right colision
 35 		if( this.rightLimit.intersects(_game.viewport.x, _game.viewport.y,
 36 			_game.viewport.width, _game.viewport.height ) ){
 37 			_game.viewport.x = this.rightLimit.x - _game.viewport.width;
 38 		}
 39 
 40 		// check left colision
 41 		if( this.leftLimit.intersects(_game.viewport.x, _game.viewport.y,
 42 			_game.viewport.width, _game.viewport.height ) ){
 43 			_game.viewport.x = this.leftLimit.x + this.leftLimit.width;
 44 		}
 45 
 46 
 47 		// check top colision
 48 		if( this.topLimit.intersects(_game.viewport.x, _game.viewport.y,
 49 			_game.viewport.width, _game.viewport.height ) ){
 50 			_game.viewport.y = this.topLimit.y + this.topLimit.height;
 51 		}
 52 
 53 		// check bottom colision
 54 		if( this.bottomLimit.intersects(_game.viewport.x, _game.viewport.y,
 55 			_game.viewport.width, _game.viewport.height ) ){
 56 			_game.viewport.y = this.bottomLimit.y - _game.viewport.height;
 57 		}
 58 
 59 	};
 60 	
 61 }
