1 /**
  2  * @class Transition
  3  * @return
  4  */
  5 function Transition(_nextScene) {
  6 
  7 	this.nextScene = _nextScene;
  8 	this.previousScreen = null;
  9 	this.nextScreen = null;
 10 
 11 	this.ready = false;
 12 	this.finished = false;
 13 	this.steps = 0;
 14 	this.currentStep = 1;
 15 	this.speed = 15;
 16 	this.expectedFPS = 60;
 17 
 18 	this.effectFunction = null;
 19 	this.EFFECT_SLIDE_LEFT = 1;
 20 	this.EFFECT_SLIDE_RIGHT = 2;
 21 	this.EFFECT_SLIDE_TOP = 3;
 22 	this.EFFECT_SLIDE_DOWN = 4;
 23 	this.EFFECT_SLIDE_FADE = 5;
 24 	this.effect = this.EFFECT_SLIDE_LEFT;
 25 
 26 	/**
 27 	 * there is a bug with firefox:
 28 	 * https://bugzilla.mozilla.org/show_bug.cgi?id=564332
 29 	 */
 30 	this.update = function(_game, current) {
 31 		if (current.effectFunction) {
 32 			current.effectFunction(_game, current);
 33 		}
 34 	};
 35 
 36 	/**
 37 	 * initial setup of the transition
 38 	 */
 39 	this.start = function(_game, current) {
 40 
 41 		// capture the current screen
 42 		if (!current.previousScreen) {
 43 			var data = _game.ctx.getImageData(0, 0, _game.width, _game.height);
 44 
 45 			current.previousScreen = document.createElement('canvas');
 46 			current.previousScreen.width = _game.width;
 47 			current.previousScreen.height = _game.height;
 48 			var localContext = current.previousScreen.getContext('2d');
 49 			localContext.putImageData(data, 0, 0);
 50 		}
 51 
 52 		if (!current.nextScreen) {
 53 			// save the first next scene screen
 54 			current.nextScreen = document.createElement('canvas');
 55 			current.nextScreen.width = _game.width;
 56 			current.nextScreen.height = _game.height;
 57 
 58 			var currentCtx = _game.ctx;
 59 			_game.ctx = current.nextScreen.getContext('2d');
 60 
 61 			// start the next scene
 62 			current.nextScene.start(_game);
 63 			current.nextScene.update(_game);
 64 
 65 			// restore game context
 66 			_game.ctx = currentCtx;
 67 		}
 68 
 69 		current.ready = true;
 70 
 71 		if (current.effect == current.EFFECT_SLIDE_LEFT) {
 72 			current.effectFunction = _effectSlideLeft;
 73 			current.steps = _game.width;
 74 		}
 75 
 76 		if (current.effect == current.EFFECT_SLIDE_RIGHT) {
 77 			current.effectFunction = _effectSlideRight;
 78 			current.steps = _game.width;
 79 		}
 80 
 81 		if (current.effect == current.EFFECT_SLIDE_TOP) {
 82 			current.effectFunction = _effectSlideTop;
 83 			current.steps = _game.height;
 84 		}
 85 
 86 		if (current.effect == current.EFFECT_SLIDE_DOWN) {
 87 			current.effectFunction = _effectSlideDown;
 88 			current.steps = _game.height;
 89 		}
 90 
 91 		if (current.effect == current.EFFECT_FADE) {
 92 			current.effectFunction = _effectFade;
 93 			current.steps = _game.height;
 94 		}
 95 	};
 96 
 97 }
 98 
 99 function _effectSlideLeft(_game, current) {
100 
101 	if (current.ready && !current.finished) {
102 		var x1 = -current.currentStep;
103 		var x2 = _game.width - current.currentStep;
104 		var localSpeed = current.speed * (current.expectedFPS / _game.lastFPS);
105 
106 		// effect to shift left
107 		_game.ctx.drawImage(current.previousScreen, x1, 0);
108 		_game.ctx.drawImage(current.nextScreen, x2, 0);
109 		current.currentStep += localSpeed;
110 	}
111 
112 	if (current.currentStep > current.steps) {
113 		current.finished = true;
114 	}
115 
116 }
117 
118 function _effectSlideRight(_game, current) {
119 
120 	if (current.ready && !current.finished) {
121 		var x1 = current.currentStep;
122 		var x2 = -_game.width + current.currentStep;
123 		var localSpeed = current.speed * (current.expectedFPS / _game.lastFPS);
124 
125 		_game.ctx.drawImage(current.previousScreen, x1, 0);
126 		_game.ctx.drawImage(current.nextScreen, x2, 0);
127 		current.currentStep += localSpeed;
128 	}
129 
130 	if (current.currentStep > current.steps) {
131 		current.finished = true;
132 	}
133 }

143 		_game.ctx.drawImage(current.nextScreen, 0, y2);

152 function _effectSlideDown(_game, current) {
153 
154 	if (current.ready && !current.finished) {
155 		var y1 = -current.currentStep;
156 		var y2 = _game.height - current.currentStep;
157 		var localSpeed = current.speed * (current.expectedFPS / _game.lastFPS);
158 
159 		_game.ctx.drawImage(current.previousScreen, 0, y1);
160 		_game.ctx.drawImage(current.nextScreen, 0, y2);
161 		current.currentStep += localSpeed;
162 	}
163 
164 	if (current.currentStep > current.steps) {
165 		current.finished = true;
166 	}
167 }
168 
169 function _effectFade(_game, current) {
170 
171 	if (current.ready && !current.finished) {
172 
173 		var half = (current.steps / 2);
174 
175 		// first part
176 		if (current.currentStep < half) {
177 			var alpha = current.currentStep / half;
178 			_game.ctx.drawImage(current.previousScreen, 0, 0);
179 			_game.ctx.fillStyle = 'rgba(0, 0, 0, ' + alpha + ')';
180 			_game.ctx.fillRect(0, 0, _game.width, _game.height);
181 		} else {
182 			var alpha = 1 - ((current.currentStep - half) / half);
183 			// second part
184 			_game.ctx.drawImage(current.nextScreen, 0, 0);
185 			_game.ctx.fillStyle = 'rgba(0, 0, 0, ' + alpha + ')';
186 			_game.ctx.fillRect(0, 0, _game.width, _game.height);
187 		}
188 
189 		var localSpeed = current.speed * (current.expectedFPS / _game.lastFPS);
190 		current.currentStep += localSpeed;
191 	}
192 
193 	if (current.currentStep > current.steps) {
194 		current.finished = true;
195 	}
196 }
