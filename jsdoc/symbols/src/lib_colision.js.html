1 function ColisionManager() {
  2 
  3 	this.sizeX = 10;
  4 	this.sizeY = 10;
  5 	this.handlers = new Array();
  6 
  7 	this.regions = new Array(10);
  8 	for ( var n = 0; n < this.sizeX; n++) {
  9 		this.regions[n] = new Array(this.sizeY);
 10 		for ( var m = 0; m < this.sizeY; m++) {
 11 			this.regions[n][m] = new Array();
 12 		}
 13 	}
 14 
 15 	/**
 16 	 * @beta
 17 	 */
 18 	this.add = function(object) {
 19 		object.regionX = Math.floor(object.x) % this.sizeX;
 20 		object.regionY = Math.floor(object.y) % this.sizeY;
 21 
 22 		object.regionX1 = Math.floor(object.x + object.width) % this.sizeX;
 23 		object.regionY1 = Math.floor(object.y + object.height) % this.sizeY;
 24 
 25 		if (object.regionX <= 0) {
 26 			object.regionX = 0;
 27 		}
 28 
 29 		if (object.regionY <= 0) {
 30 			object.regionY = 0;
 31 		}
 32 
 33 		game.debug.log('regions ' + object.regionX + ' ' + object.regionY + ' '
 34 				+ object.x + ' ' + object.y);
 35 
 36 		this.regions[object.regionX][object.regionY].push(object);
 37 	};
 38 
 39 
 40 	/**
 41 	 * @beta
 42 	 */
 43 	this.move = function(object) {
 44 
 45 		// remove from the list
 46 		for (key in this.regions[object.regionX][object.regionY]) {
 47 			if (object.id == this.regions[object.regionX][object.regionY][key].id) {
 48 				delete this.regions[object.regionX][object.regionY][key];
 49 				break;
 50 			}
 51 		}
 52 
 53 		this.add(object);
 54 	};
 55 
 56 	// dispatch the colision for the listeners
 57 	this.colide = function(_game, _a, _b) {
 58 
 59 		for (key in this.handlers) {
 60 			handler = this.handlers[key];
 61 			var result = handler(_game, _a, _b);
 62 			if (result) {
 63 				_game.debug.log('colision processing interrupted.');
 64 				return true;
 65 			}
 66 		}
 67 	};
 68 
 69 	this.update = function(_game) {
 70 		this.checkColision(_game);
 71 	};

 89 									neighb1[key].y, neighb1[key].width,
 90 									neighb1[key].height)) {
 91 								this.colide(_game, childs[n], neighb1[key]);
 92 							}
 93 						}
 94 					}
 95 				}

102

113 								}

116 					}
117 				}
118

121

168 	this.checkColision = function(_game) {
169 
170 		var childs = _game.currentScene.childs;
171 		for ( var n = 0; n < childs.length; n++) {
172 			if (childs[n].isColidable) {
173 
174 				// all the childs before the current
175 				var before1 = 0;
176 				var before2 = n - 1;
177 				for ( var i = before1; i < before2; i++) {
178 					if (childs[i].isColidable) {
179 						if (childs[n].intersects(childs[i].x, childs[i].y,
180 								childs[i].width, childs[i].height)) {
181 							this.colide(_game, childs[n], childs[i]);
182 						}
183 					}
184 				}
185 
186 				// all the childs after the current
187 				var after1 = n + 1;
188 				var after2 = childs.length;
189 				for ( var i = after1; i < after2; i++) {
190 					if (childs[i].isColidable) {
191 						if (childs[n].intersects(childs[i].x, childs[i].y,
192 								childs[i].width, childs[i].height)) {
193 							this.colide(_game, childs[n], childs[i]);
194 						}
195 					}
196 				}
197 			}
198 
199 		}
200 
201 	};
