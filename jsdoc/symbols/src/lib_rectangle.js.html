1 /**
  2  * @class Rectangle
  3  * @return
  4  */
  5 function Rectangle(_id, _x, _y, _width, _height) {
  6 
  7 	this.id = _id;
  8 	this.x = _x;
  9 	this.y = _y;
 10 	this.width = _width;
 11 	this.height = _height;
 12 	
 13 	this.remove = false;
 14 	
 15 	// by pass the viewport
 16 	this.fixed = false;
 17 
 18 	// drag and drop control
 19 	this.dragX = -1;
 20 	this.dragY = -1;
 21 	this.isDragging = false;
 22 	this.isDraggable = false;
 23 
 24 	// for the colision
 25 	this.isColidable = false;
 26 
 27 	this.debug = false;
 28 
 29 	this.fillColor = NMSColor.AVOCADO;
 30 	this.borderColor = NMSColor.BLUEBERRY;
 31 	this.border = 2;
 32 
 33 	this.pointerDownCallbacks = new Array();
 34 	this.pointerUpCallbacks = new Array();
 35 	this.pointerMoveCallbacks = new Array();
 36 
 37 	this.pointerDown = function(_game, x, y, _child) {
 38 		for (key in this.pointerDownCallbacks) {
 39 			callback = this.pointerDownCallbacks[key];
 40 			if (callback) {
 41 				var result = callback(_game, x, y, _child);
 42 
 43 				// if return true interrupt the process
 44 				if (result) {
 45 					_game.debug
 46 							.log('event consumed - interrupt processing of pointerdown');
 47 					return true;
 48 				}
 49 			}
 50 		}
 51 	};
 52 
 53 	this.pointerUp = function(_game, x, y, _child) {
 54 		for (key in this.pointerUpCallbacks) {
 55 			callback = this.pointerUpCallbacks[key];
 56 			if (callback) {
 57 				var result = callback(_game, x, y, _child);
 58 
 59 				// if return true interrupt the process
 60 				if (result) {
 61 					_game.debug
 62 							.log('event consumed - interrupt processing of pointerup');
 63 					return true;
 64 				}
 65 			}
 66 		}
 67 	};
 68 
 69 	this.pointerMove = function(_game, x, y, _child) {
 70 		for (key in this.pointerMoveCallbacks) {
 71 			callback = this.pointerMoveCallbacks[key];
 72 			if (callback) {
 73 				var result = callback(_game, x, y, _child);
 74 
 75 				// if return true interrupt the process
 76 				if (result) {
 77 					_game.debug
 78 							.log('event consumed - interrupt processing of pointermove (2)');
 79 					return true;
 80 				}
 81 			}
 82 		}
 83 	};
 84 
 85 	// control drag and drop
 86 	this.pointerDownCallbacks.push(function(_game, x, y, _child) {
 87 		_child.dragX = x;
 88 		_child.dragY = y;
 89 
 90 		if (_child.isDraggable) {
 91 			_child.isDragging = true;
 92 		}
 93 		return true;
 94 	});
 95 
 96 	this.drag = function(_game, x, y, _child) {
 97 
 98 		if (this.debug) {
 99 			_game.debug.log('moved ' + x + '/' + y + ' drag:'
100 					+ _child.isDraggable + ' dragging:' + _child.isDragging
101 					+ ' id:' + _child.id);
102 		}
103 
104 		if (_child.isDraggable && _child.isDragging) {
105 			_child.x = _child.x + (x - _child.dragX);
106 			_child.y = _child.y + (y - _child.dragY);
107 			_child.dragX = x;
108 			_child.dragY = y;
109 			return true;
110 		}
111 	};
112 
113 	/**
114 	 * should be override by implementation
115 	 */
116 	this.beforeUpdate = function(_game, _scene, current) {
117 		// empty
118 	}
119 	
120 	this.update = function(_game, _scene, current) {
121 		this.beforeUpdate(_game, _scene, current);
122 		
123 		if( this.width < 1 || this.height < 1 ){
124 			_game.debug.warn('update of not loaded : ' 
125 			+ this.id + ' ' + this.width + ' ' + this.height );
126 			return;
127 		}
128 		
129 		if( this.fixed ){
130 			var ctx = _game.ctx;
131 			ctx.strokeStyle = this.borderColor;
132 			ctx.lineWidth = this.border;
133 			ctx.fillStyle = this.fillColor;
134 			
135 			var _x = this.x;
136 			var _y = this.y;
137 			
138 			ctx.beginPath();
139 			ctx.moveTo( _x, _y);
140 			ctx.lineTo( _x + this.width, _y);
141 			ctx.lineTo( _x + this.width, _y + this.height);
142 			ctx.lineTo( _x, _y + this.height);
143 			ctx.lineTo( _x, _y );
144 	
145 			ctx.fill();
146 			ctx.stroke();
147 			ctx.closePath();
148 		
149 		} else {
150 			if( this.intersects(_game.viewport.x, _game.viewport.y,
151 				_game.viewport.width, _game.viewport.height ) ){
152 				
153 				var ctx = _game.ctx;
154 				ctx.strokeStyle = this.borderColor;
155 				ctx.lineWidth = this.border;
156 				ctx.fillStyle = this.fillColor;
157 				
158 				var _x = this.x - _game.viewport.x;
159 				var _y = this.y - _game.viewport.y;
160 				
161 				ctx.beginPath();
162 				ctx.moveTo( _x, _y);
163 				ctx.lineTo( _x + this.width, _y);
164 				ctx.lineTo( _x + this.width, _y + this.height);
165 				ctx.lineTo( _x, _y + this.height);
166 				ctx.lineTo( _x, _y );
167 		
168 				ctx.fill();
169 				ctx.stroke();
170 				ctx.closePath();
171 			} 
172 		}
173 	};
174 
175 	this.has = function has(x, y) {
176 		return (x > this.x) && (y > this.y) && (x < this.x + this.width)
177 				&& (y < this.y + this.height);
178 	};
179 
180 	this.intersects = function(x1, y1, width1, height1) {
181 
182 		var x2 = this.x;
183 		var y2 = this.y;
184 		var width2 = this.width;
185 		var height2 = this.height;
186 
187 		return !((x2 > (x1 + width1)) || (x1 > (x2 + width2))
188 				|| (y2 > (y1 + height1)) || (y1 > (y2 + height2)));
189 
190 	};
191 
192 }
