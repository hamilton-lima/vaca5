1 /**
  2  * @class Label
  3  * @param _x
  4  * @param _y
  5  * @param _id
  6  * @param _text
  7  * @return
  8  */
  9 function Label(_x, _y, _id, _text) {
 10 
 11 	this.from = Rectangle;
 12 	this.from();
 13 
 14 	this.x = _x;
 15 	this.y = _y;
 16 	this.id = _id;
 17 	this.text = _text;
 18 	this.size = 20;
 19 	this.font = 'Helvetica';
 20 	this.debug = false;
 21 	this.invalid = true;
 22 
 23 	this.innerBorder = 2;
 24 	this.align = 'center';
 25 	this.valign = 'center';
 26 	
 27 	this.textWidth = null;
 28 	this.textHeight = null;	
 29 
 30 	this.fillColor = NMSColor.EGGPLANT;
 31 	this.borderColor = NMSColor.BLUEBERRY;
 32 	this.backgroundColor = null;
 33 	
 34 	this.border = 1;
 35 
 36 	/**
 37 	 * activate debug mode of the label, show a rectangle around the label
 38 	 */
 39 	this.setDebug = function(_bool) {
 40 		this.debug = _bool;
 41 	};	
 42 	
 43 	/**
 44 	 * change the text and recalculate the dimensions
 45 	 */
 46 	this.setText = function(_text) {
 47 		this.text = _text;
 48 		this.invalid = true;
 49 	};
 50 	
 51 	this.calculateMetrics = function(_game){
 52 		_game.ctx.font = this.size + 'px ' + this.font;
 53 		_game.ctx.textBaseline = 'top';
 54 		var metrics = _game.ctx.measureText(this.text);
 55 		this.textWidth = metrics.width;
 56 		this.textHeight = this.size;
 57 		
 58 		if( ! this.width ){
 59 			this.width = this.textWidth;
 60 		}		
 61 		
 62 		if( ! this.height ){
 63 			this.height = this.textHeight;
 64 		}
 65 		
 66 		if (this.debug) {
 67 			_game.debug.log(this.id + ' font bounding box : ' + this.width
 68 					+ '/' + this.height);
 69 		}
 70 
 71 	};
 72 	
 73 	this.start = function(_game) {
 74 		this.calculateMetrics(_game);
 75 	};
 76 	
 77 	
 78 	this.beforeUpdate = function(_game, _scene, current){
 79 		
 80 	};
 81 	
 82 	this.update = function(_game, _scene, current) {
 83 		this.beforeUpdate(_game, _scene, current);
 84 		
 85 		if( this.width <= 0 || this.height <= 0 ){
 86 			_game.debug.warn('update of not loaded : ' + this.id );
 87 			return;
 88 		}
 89 		
 90 		if( this.invalid ){
 91 			this.calculateMetrics(_game);
 92 		}
 93 
 94 		var ctx = _game.ctx;
 95 
 96 		if( current.backgroundColor ){
 97 			_game.ctx.fillStyle = current.backgroundColor;
 98 			_game.ctx.beginPath();
 99 			_game.ctx.rect(this.x, this.y, this.width, this.height);
100 			_game.ctx.closePath();
101 			_game.ctx.fill();	
102 		}
103 		
104 		ctx.strokeStyle = this.borderColor;
105 		ctx.lineWidth = this.border;
106 		ctx.fillStyle = this.fillColor;
107 
108 		ctx.textBaseline = 'top';
109 		ctx.font = this.size + 'px ' + this.font;
110 		
111 		var localX = this.x;
112 		var localY = this.y;
113 		
114 		if( this.align == 'center'){
115 			localX = localX + ((this.width - this.textWidth ) /2 );
116 		} else {
117 			localX = localX + this.innerBorder;
118 		}
119 		
120 		if( this.valign == 'center'){
121 			localY = localY + ((this.height - this.textHeight ) /2 );
122 		} else {
123 			localY = localY + this.innerBorder;
124 		}
125 		
126 		if( this.fixed ){
127 			ctx.fillText(this.text, localX, localY);
128 			
129 			if (this.debug) {
130 				ctx.strokeRect(localX, localY, this.width, this.height);
131 			}
132 			
133 		} else {
134 			if( this.intersects(_game.viewport.x, _game.viewport.y,
135 				_game.viewport.width, _game.viewport.height ) ){
136 	
137 				localX -= _game.viewport.x;
138 				localY -= _game.viewport.y;
139 				
140 				ctx.fillText(this.text, localX, localY);
141 				
142 				if (this.debug) {
143 					ctx.strokeRect(localX, localY, this.width, this.height);
144 				}
145 			} 
146 		}
147 		// ctx.strokeText(this.text, this.x, this.y);
148 		
149 	};
150 
151 }
