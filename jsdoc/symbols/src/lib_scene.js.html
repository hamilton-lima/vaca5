1 /**
  2  * @class Scene
  3  * @param _width
  4  * @param _height
  5  * @return
  6  */
  7 function Scene(_width,_height){
  8 	
  9 	this.x = 0;
 10 	this.y = 0;
 11 	
 12 	this.width = _width;
 13 	this.height = _height;
 14 	
 15 	this.childs = new Array();
 16 	this.background = '#000000';
 17 	
 18 	this.pointerDownCallbacks = new Array();
 19 	this.pointerUpCallbacks = new Array();
 20 	this.pointerMoveCallbacks = new Array();
 21 	
 22 	this.transition = null;
 23 	
 24 	// trigger clicks on scene childs
 25 	this.pointerDownCallbacks.push(function(_game, x, y) {
 26 		
 27 		var childs = _game.currentScene.childs;
 28 		
 29 		for(var n = childs.length-1; n >=0; n--){
 30 			if( childs[n].has && childs[n].has(x,y)){
 31 				// when event is consumed the method returns true.
 32 				if( childs[n].pointerDown(_game,x,y,childs[n])){
 33 					break;
 34 				}
 35 			}
 36 		}
 37 	});
 38 	
 39 	this.pointerUpCallbacks.push(function(_game, x, y) {
 40 		
 41 		var childs = _game.currentScene.childs;
 42 		
 43 		for(var n = childs.length-1; n >=0; n--){
 44 			if( childs[n].has && childs[n].has(x,y)){
 45 				// when event is consumed the method returns true.
 46 				if( childs[n].pointerUp(_game,x,y,childs[n])){
 47 					break;
 48 				}
 49 			}
 50 		}
 51 
 52 		// release the mouse on all childrens
 53 		for(key in childs){
 54 			child = childs[key]; 
 55 			child.isDragging = false;
 56 		}
 57 	});
 58 	
 59 	this.pointerMoveCallbacks.push(function(_game, x, y) {
 60 		
 61 		var childs = _game.currentScene.childs;
 62 		
 63 		// move on dragging child
 64 		for(key in childs){
 65 			child = childs[key]; 
 66 			if( child.isDragging ){
 67 				child.drag(_game,x,y,child);
 68 			}
 69 		}
 70 		
 71 		// other mouse move events
 72 		for(var n = childs.length-1; n >=0; n--){
 73 			if( childs[n].has && childs[n].has(x,y)){
 74 				// when event is consumed the method returns true.
 75 				if( childs[n].pointerMove(_game,x,y,childs[n])){
 76 					break;
 77 				}
 78 			}
 79 		}
 80 		
 81 		
 82 	});	
 83 	
 84 	this.beforeUpdate = function(_game, current) {
 85 		// empty
 86 	};	
 87 	
 88 	/**
 89 	 * called by the game main loop
 90 	 */
 91 	this.update = function(_game, current) {
 92 		this.beforeUpdate(_game, current);
 93 		this.clearBackGround(_game.ctx);
 94 		var child = null;
 95 		for(key in this.childs){
 96 			child = this.childs[key];
 97 			try {
 98 				if( child.update){
 99 					child.update(_game, current, child);
100 				}
101 			} catch(updateError){
102 				_game.debug.error('error on update : ' + updateError );
103 			}
104 		}
105 
106 		// childs cleanup	
107 		for(var n = 0; n < this.childs.length; n++){
108 			child = this.childs[n];
109 			
110 			if(child.remove){
111 				this.childs.splice(n,1);
112 			}
113 		}
114 		
115 	};
116 
117 	/**
118 	 * call the start method of all child, if the child has start method
119 	 */
120 	this.start = function(_game) {
121 		_game.debug.log('scene start');
122 		var child = null;
123 		for(key in this.childs){
124 			child = this.childs[key]; 
125 
126 			if(child.start){
127 				child.start(_game, child);
128 			}
129 		}
130 	};
131 
132 	this.clearBackGround = function(ctx){
133 		ctx.fillStyle = this.background;
134 		ctx.beginPath();
135 		ctx.rect(this.x, this.y, this.width, this.height);
136 		ctx.closePath();
137 		ctx.fill();		
138 	};
139 	
140 	this.setBackground = function(_background){
141 		this.background = _background;
142 	};
143 	
144 	this.add = function(child){
145 		this.childs.push(child);
146 	};
147 	
148 	this.findById = function(_id){
149 		var child = null;
150 		for( var key in this.childs){
151 			child = this.childs[key];
152 			if( child.id == _id ){
153 				return child;
154 			}
155 		}
156 		return null;
157 	};
158 	
159 	/**
160 	 * nickname for findById()
161 	 */
162 	this.find = this.findById;
163 	
164 }

166 function SceneEmpty(_width,_height){
167 	this.from = Scene;
168 	this.from();
169 	var label = new Label(10,30,'_title','Empty scene.');
170 	this.add(label);
171 	this.setBackground(NMSColor.BANANA);
172 
173 	this.width = _width;
174 	this.height = _height;
175 }

177 function SceneLoading(_width, _height, _nextScene ){
178 	this.from = Scene;
179 	this.from();
180 
181 	this.setBackground('#000000');	
182 	this.nextScene = _nextScene;
183 	
184 	var w = _width * 0.9;
185 	var x = (_width - w) / 2;
186 	var y = _height - 20;
187 	
188 	this.barWidth = w;
189 	
190 	this.start = function(_game,current) {
191 		var label = new Label(0,0,'_title_loading','Loading');
192 		label.calculateMetrics(_game);
193 		label.x = ( _game.width - label.width ) / 2;
194 		label.y = ( _game.height - label.height ) / 2; 
195 		
196 		label.fillColor = '#CCCCCC';
197 		label.borderColor = '#CCCCCC';
198 
199 		current.add(label);	
200 		
201 		var backRect = new Rectangle('back_loading', x, y, w, 5);	
202 		var frontRect = new Rectangle('front_loading', x, y, 1, 5);
203 		
204 		backRect.fillColor = '#CCCCCC';
205 		backRect.borderColor = '#CCCCCC';
206 		frontRect.fillColor = NMSColor.PAPAYA;
207 		frontRect.borderColor = NMSColor.PAPAYA;
208 		
209 		current.add(backRect);	
210 		current.add(frontRect);
211 		
212 		var front = current.findById('front_loading');
213 		front.width = current.barWidth * _resourceManager.percentage();
214 	};	
215 	
216 	this.beforeUpdate = function(_game,current) {
217 		var front = current.findById('front_loading');
218 		var percent = _resourceManager.percentage();
219 		
220 		front.width = current.barWidth * percent;
221 		if( front.width < 1 ){
222 			front.width = 1;
223 		}
224 		
225 		if( percent >= 1 ){
226 			current.transition =  new Transition(current.nextScene);
227 			current.transition.effect = current.transition.EFFECT_FADE;
228 		}
229 	};	
230 
231 
232 	this.width = _width;
233 	this.height = _height;
234 };
