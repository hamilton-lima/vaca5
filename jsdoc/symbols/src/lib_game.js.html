  3 /**
  4  * @class Game
  5  * @param _width
  6  * @param _height

  9 function Game(_width, _height) {
 10 	this.VERSION = '20120109-animate';
 11 	
 12 	this.width = _width;
 13 	this.height = _height;
 14 
 15 	this.debug = new Debug('debug_div');
 16 	this.audio = new AudioManager(this);
 17 
 18 	this.ctx = null;
 19 	this.canvas = null;
 20 	this.currentScene = new SceneEmpty(this.width, this.height);
 21 	this.supportsTouch = false;
 22 	
 23 	this.VIEWPORT_NAME = '_viewport';
 24 	this.viewport = new Rectangle(this.VIEWPORT_NAME, 0, 0, _width, _height);
 25 
 26 	this.browser = null;
 27 	this.version = null;
 28 	this.osName = null;
 29 
 30 	this.frameCounter = 0;
 31 	this.lastFPS = 0;
 32 	this.showFPS = true;
 33 	this.fpsColor = '#000000';
 34 
 35 	this.keyPressed = new Array();
 36 	this.keyMapping = _defaultKeyMapping();
 37 	this.colision = new ColisionManager();
 38 
 39 	this.interval = 1;
 40 	this.dt = 0;
 41 	this.now = timestamp(); 
 42 	this.last = timestamp();
 43 
 44 	this.keys = new Array();
 45 	
 46 	/**
 47 	 * support for language resources
 48 	 */
 49 	this.s = function(key,value){
 50 		return this.keys[key];
 51 	}
 52 
 53 	this.def = function(key,value){
 54 		this.keys[key] = null;
 55 		this.keys[key] = value;
 56 	}
 57 	
 58 	/**
 59 	 * game main loop
 60 	 * 
 61 	 * @return
 62 	 */
 63 	this.update = function() {
 64 		this.frameCounter++;
 65 
 66 		if (!this.currentScene.transition) {
 67 			this.colision.update(this);
 68 			this.currentScene.update(this, this.currentScene);
 69 
 70 			if (this.showFPS) {
 71 				this.ctx.font = '15px helvetica';
 72 				this.ctx.fillStyle = this.fpsColor;
 73 				this.ctx.fillText(this.lastFPS + 'FPS', 10, 10);
 74 			}
 75 		} else {
 76 
 77 			if (!this.currentScene.transition.ready) {
 78 				this.currentScene.transition.start(this,
 79 						this.currentScene.transition);
 80 			} else {
 81 				this.currentScene.transition.update(this,
 82 						this.currentScene.transition);
 83 			}
 84 
 85 			// remove transition after it finishes
 86 			if (this.currentScene.transition.finished) {
 87 				var sceneTemp = this.currentScene;
 88 				this.currentScene = sceneTemp.transition.nextScene;
 89 				this.currentScene.start(this);
 90 				sceneTemp.transition = null;
 91 			}
 92 		}
 93 
 94 	};
 95 
 96 	this.start = function() {
 97 		this.ctx = this.canvas.getContext('2d');
 98 		this.currentScene.start(this, this.currentScene);
 99 
100 		_update();
101 		_FPSCounter();
102 	};
103 
104 	this.updateFPS = function() {
105 		this.lastFPS = this.frameCounter;
106 		this.frameCounter = 0;
107 	};
108 
109 	// check the key codes based on the mapping
110 	// and look for the keys state in the keyPressed array
111 	//
112 	this.isKeyPressed = function(name) {
113 
114 		var target = null;
115 		for (key in this.keyMapping) {
116 
117 			var mapping = this.keyMapping[key];
118 			if (mapping.name == name) {
119 				target = mapping.keys;
120 				break;
121 			}
122 		}
123 
124 		this.debug.log('isKeyPressed() target: ' + target);
125 
126 		var result = false;
127 		if (target) {
128 			for (key in target) {
129 				if (this.keyPressed[target[key]]) {
130 					result = true;
131 					break;
132 				}
133 			}
134 
135 		} else {
136 			this.debug.log('key: [' + name + '] not assigned.');
137 		}
138 		return result;
139 	};
140 
141 }

163 function _update() {
164     game.now = timestamp();
165     game.dt = (game.now - game.last) / 100;
166 
167 	game.update();
168 	game.last = game.now;
169 
170 	setTimeout(_update, game.interval);

197

199 	_canvas.height = _height;

232 function _mouseDown(e) {
233 
234 	// disable mouse input during the transition
235 	if (game.currentScene.transition) {
236 		return;
237 	}
238 
239 	var touches = getTouches(e);
240 
241 	for (n in touches) {
242 		var p = touches[n];
243 
244 		var callback = null;
245 		for (key in game.currentScene.pointerDownCallbacks) {
246 			callback = game.currentScene.pointerDownCallbacks[key];
247 			if (callback) {
248 				callback(game, p.x, p.y);
249 			}
250 		}
251 	}
252 
253 }

255 function _mouseUp(e) {
256 	
257 	// disable mouse input during the transition
258 	if (game.currentScene.transition) {
259 		return;
260 	}
261 
262 	var touches = getTouches(e);
263 
264 	// on ipad there is no touches when the pointer is up
265 	if (touches.length <= 0) {
266 		for (key in game.currentScene.pointerUpCallbacks) {
267 			callback = game.currentScene.pointerUpCallbacks[key];
268 			if (callback) {
269 				callback(game);
270 			}
271 		}
272 	} else {
273 		for (n in touches) {
274 			var p = touches[n];
275 
276 			var callback = null;
277 			for (key in game.currentScene.pointerUpCallbacks) {
278 				callback = game.currentScene.pointerUpCallbacks[key];
279 				if (callback) {
280 					callback(game, p.x, p.y);
281 				}
282 			}
283 		}

315 function getTouches(e) {
316 
317 	var result = new Array();
318 
319 	if (e.touches) {
320 		// Touch Enabled (loop through all touches)
321 		for ( var i = 1; i <= e.touches.length; i++) {
322 			var p = getCoords(e.touches[i - 1]); // Get info for finger i
323 			result.push(p);
324 		}
325 	} else {
326 		// Not touch enabled (get cursor position from single event)
327 		var p = getCoords(e);
328 		result.push(p);
329 	}
330 
331 	return result;
332 }

334 function getCoords(e) {
335 	if (e.offsetX) {
336 		// Works in Chrome / Safari (except on iPad/iPhone)
337 		return {
338 			x : e.offsetX,
339 			y : e.offsetY
340 		};
341 	} else if (e.layerX) {
342 		// Works in Firefox
343 		return {
344 			x : e.layerX,
345 			y : e.layerY
346 		};
347 	} else {
348 		// Works in Safari on iPad/iPhone
349 		return {
350 			x : e.pageX - game.canvas.offsetLeft,
351 			y : e.pageY - game.canvas.offsetTop
352 		};
353 	}
