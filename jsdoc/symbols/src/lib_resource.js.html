  3 function ResourceManager() {
  4 
  5 	this.images = new Array();
  6 	this.sounds = new Array();
  7 	this.active = false;
  8 
  9 	this.imagesCallbacks = new Array();
 10 	this.soundsCallbacks = new Array();
 11 	this.interval = 50;
 12 
 13 	/**
 14 	 * 
 15 	 * @return image resources loading percentage
 16 	 */
 17 	this.percentage = function(){
 18 
 19 		var lasting = 0;
 20 		for(key in _resourceManager.imagesCallbacks){
 21 			lasting ++;
 22 		}
 23 		
 24 		var total = 0;
 25 		for(key in _resourceManager.images){
 26 			total ++;
 27 		}
 28 		
 29 		var current = total - lasting;
 30 		var percentage = current / total;
 31 		
 32 		if( total == 0 ){
 33 			percentage = 0;
 34 		}
 35 		
 36 		return percentage;
 37 	};
 38 	
 39 
 40 	/**
 41 	 * @param src,
 42 	 *            the image source name
 43 	 * @param callback,
 44 	 *            the object that will receive the call of the method
 45 	 *            callback(object,image); when the onload is triggered.
 46 	 */
 47 	this.addImage = function(src, callback) {
 48 				
 49 		var _image = new Image();
 50 		_image.src = src;
 51 
 52 		if (!this.imagesCallbacks[src]) {
 53 			this.imagesCallbacks[src] = new Array();
 54 		}
 55 
 56 		this.imagesCallbacks[src].push(callback);
 57 		this.images[src] = _image;
 58 
 59 		if (!this.active) {
 60 			this.active = true;
 61 			_resourceManager_update();
 62 		}
 63 
 64 	};
 65 
 66 }
 67 
 68 var _resourceManager = new ResourceManager();
 69 
 70 function _resourceManager_update() {
 71 
 72 	for (src in _resourceManager.images) {
 73 
 74 		var callbacks = _resourceManager.imagesCallbacks[src];
 75 		if (callbacks) {
 76 
 77 			if (_resourceManager.images[src].complete 
 78 				&& _resourceManager.images[src].width > 0				
 79 				&& _resourceManager.images[src].height > 0 ) {
 80 
 81 				for (target in callbacks) {
 82 					var _target = callbacks[target];
 83 					_target.callback(_target, _resourceManager.images[src]);
 84 				}
 85 				delete _resourceManager.imagesCallbacks[src];
 86 			}
 87 		}
 88 	}
 89 	
 90 	var count = 0;
 91 	for(key in _resourceManager.imagesCallbacks){
 92 		count ++;
 93 	}
 94 
 95 	if (count > 0) {
 96 		setTimeout(_resourceManager_update, _resourceManager.interval);
 97 	} else {
 98 		_resourceManager.active = false;
 99 	}
100 }
