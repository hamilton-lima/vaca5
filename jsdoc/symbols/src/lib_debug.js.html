  6 /**
  7  * @class Debug
  8  * @param divName
  9  * @return
 10  */
 11 function Debug(divName) {
 12 
 13 	this.ERROR = 1;
 14 	this.WARN = 2;
 15 	this.LOG = 3;
 16 	this.INFO = 4;
 17 	
 18 	this.limit = 15;
 19 	this.debugDiv = document.getElementById(divName);
 20 	this.messages = new Array();
 21 	this.level = this.WARN;
 22 
 23 	this.error = function(_text) {
 24 		this.addMessage(this.ERROR,_text);
 25 	};
 26 	
 27 	this.warn = function(_text) {
 28 		this.addMessage(this.WARN,_text);
 29 	};
 30 
 31 	this.log = function(_text) {
 32 		this.addMessage(this.LOG,_text);
 33 	};
 34 
 35 	this.info = function(_text) {
 36 		this.addMessage(this.INFO,_text);
 37 	};
 38 
 39 	this.addMessage = function(_level, _text) {
 40 		if( this.level < _level ){
 41 			return;
 42 		}
 43 		
 44 		var message = new DebugMessage(_level, _text);
 45 		this.messages.unshift(message);
 46 		this.show();
 47 	};
 48 	
 49 	this.levelLabel = function(_level){
 50 		var result = '#N/D';
 51 		switch(_level){
 52 		case 1:
 53 			  result = 'ERROR';
 54 			  break;
 55 		case 2:
 56 			  result = 'WARN';
 57 			  break;
 58 		case 3:
 59 			  result = 'LOG';
 60 			  break;
 61 		case 4:
 62 			  result = 'INFO';
 63 			  break;
 64 		}
 65 		
 66 		return result;
 67 	};
 68 	
 69 	this.show = function() {
 70 
 71 		if (this.messages.length > this.limit) {
 72 			this.messages.pop();
 73 		}
 74 
 75 		if (this.debugDiv) {
 76 			var tempString = '<b>debug</b><br>';
 77 			for (key in this.messages) {
 78 				var message = this.messages[key];
 79 				if( this.level >= message.level ){
 80 					tempString += this.levelLabel(message.level) +' '+ message.text+ '<br>';
 81 				}
 82 			}
 83 			this.debugDiv.innerHTML = tempString;
 84 		}
 85 	};
 86 	
 87 	
 88 }
